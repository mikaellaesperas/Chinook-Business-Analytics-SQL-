/* PROJECT: Chinook SQL Business Analytics
   PURPOSE: Practice SQL for business insights
   DATABASE: Chinook (Music Store Dataset) */


/* 1. BASIC DATA EXPLORATION
   PURPOSE: Check total number of customers */

SELECT COUNT(*) AS TotalCustomers
FROM Customer;


/* 2. CUSTOMER DISTRIBUTION BY COUNTRY
   PURPOSE: Understand where customers are located */

SELECT 
    Country,
    COUNT(*) AS TotalCustomers
FROM Customer
GROUP BY Country
ORDER BY TotalCustomers DESC;


/* 3. CUSTOMER ACTIVITY (NUMBER OF ORDERS)
   PURPOSE: Identify most active customers by order count */

SELECT 
    c.FirstName,
    c.LastName,
    c.Country,
    COUNT(i.InvoiceId) AS TotalOrders
FROM Customer c
JOIN Invoice i
    ON c.CustomerId = i.CustomerId
GROUP BY c.CustomerId
ORDER BY TotalOrders DESC;


/* 4. CUSTOMER REVENUE ANALYSIS
   PURPOSE: Find which customers generate the most revenue */

SELECT 
    c.FirstName,
    c.LastName,
    c.Country,
    SUM(i.Total) AS TotalSpent
FROM Customer c
JOIN Invoice i
    ON c.CustomerId = i.CustomerId
GROUP BY c.CustomerId
ORDER BY TotalSpent DESC;


/* 5. TOTAL COMPANY REVENUE
   PURPOSE: Calculate total sales revenue */

SELECT 
    SUM(Total) AS TotalRevenue
FROM Invoice;


/* 6. AVERAGE ORDER VALUE
   PURPOSE: Understand typical customer spending per order */

SELECT 
    AVG(Total) AS AverageOrderValue
FROM Invoice;


/* 7. TOP SELLING TRACKS (PRODUCT ANALYSIS)
   PURPOSE: Identify most purchased music tracks */

SELECT 
    t.Name AS TrackName,
    SUM(il.Quantity) AS TotalSold
FROM InvoiceLine il
JOIN Track t 
    ON il.TrackId = t.TrackId
GROUP BY t.TrackId
ORDER BY TotalSold DESC;


/* 8. REVENUE PER TRACK
   PURPOSE: Identify highest earning tracks */

SELECT 
    t.Name AS TrackName,
    SUM(il.UnitPrice * il.Quantity) AS Revenue
FROM InvoiceLine il
JOIN Track t 
    ON il.TrackId = t.TrackId
GROUP BY t.TrackId
ORDER BY Revenue DESC;


/* 9. TOP GENRES BY REVENUE
   PURPOSE: Understand which music genres perform best */

SELECT 
    g.Name AS Genre,
    SUM(il.UnitPrice * il.Quantity) AS Revenue
FROM InvoiceLine il
JOIN Track t 
    ON il.TrackId = t.TrackId
JOIN Genre g 
    ON t.GenreId = g.GenreId
GROUP BY g.GenreId
ORDER BY Revenue DESC;


/* 10. EMPLOYEE PERFORMANCE (SALES SUPPORT)
   PURPOSE: See which employee handles most customers/orders */

SELECT 
    e.FirstName,
    e.LastName,
    COUNT(c.CustomerId) AS TotalCustomersHandled
FROM Employee e
JOIN Customer c 
    ON e.EmployeeId = c.SupportRepId
GROUP BY e.EmployeeId
ORDER BY TotalCustomersHandled DESC;
